    libvic20cart
    PRIVATE
        $<$<COMPILE_LANGUAGE:C>:HAVE_CONFIG_H NDEBUG _arch_dreamcast __DREAMCAST__ >

        $<$<COMPILE_LANGUAGE:OBJC>:>
    )
