102inline static void raster_changes_apply(raster_changes_t *changes,
103 const unsigned int idx)
104{
105 raster_changes_action_t *action;
106
107 action = changes->actions + idx;
108
109 switch (changes->actions[idx].type) {
110 case RASTER_CHANGES_TYPE_INT:
111 *action->value.integer.oldp = action->value.integer.newone;
112 break;
113 case RASTER_CHANGES_TYPE_PTR:
114 default: /* To be faster. */
115 *action->value.ptr.oldp = action->value.ptr.newone;
116 break;
117 }
118}

121inline static void raster_changes_remove_all(raster_changes_t *changes)
122{
123 changes->count = 0;
124}

127inline static void raster_changes_apply_all(raster_changes_t *changes)
128{
129 unsigned int i;
130
131 for (i = 0; i < changes->count; i++) {
132 raster_changes_apply(changes, i);
133 }
134
135 raster_changes_remove_all(changes);
136}

155inline static void raster_changes_add_sorted_int(raster_changes_t *changes,
156 const int where,
157 int *ptr,
158 const int new_value)
159{
160 raster_changes_action_t *action;
161 int j, i = (int)changes->count - 1;
162
163 while (i >= 0 && changes->actions[i].where > where) {
164 i--;
165 }
166 for (j = (int)changes->count - 1; j > i; j--) {
167 changes->actions[j + 1] = changes->actions[j];
168 }
169
170 changes->count++;
171 action = changes->actions + 1 + i;
172
173 action->where = where;
174 action->type = RASTER_CHANGES_TYPE_INT;
175 action->value.integer.oldp = ptr;
176 action->value.integer.newone = new_value;
177}
