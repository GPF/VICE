27/* This file contains the exported interface to the iec488 emulator.
28 * The iec488 emulator then calls (modifed) routines from serial.c
29 * to use the standard floppy interface.
30 * The current state of the bus and methods to set output lines
31 * are exported.
32 * This hardware emulation is necessary, as different PET kernels would
33 * need different traps. But it's also much faster than the (hardware
34 * simulated) serial bus, as it's parallel. So we need no traps.
35 */
36

83#define PARALLEL_SET_LINE(line, dev, mask) \
84 static inline void parallel_##dev##_set_##line(uint8_t val) \
85 { \
86 if (val) { \
87 parallel_set_##line(PARALLEL_##mask); \
88 } else { \
89 parallel_clr_##line(~PARALLEL_##mask); \
90 } \
91 }
92
93#define PARALLEL_RESTORE_LINE(line, dev, mask) \
94 static inline void parallel_##dev##_restore_##line(uint8_t val) \
95 { \
96 if (val) { \
97 parallel_restore_set_##line(PARALLEL_##mask); \
98 } else { \
99 parallel_restore_clr_##line(~PARALLEL_##mask); \
100 } \
101 }
