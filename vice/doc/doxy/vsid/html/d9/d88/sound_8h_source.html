224#ifndef SOUND_SYSTEM_FLOAT
225static inline int16_t sound_audio_mix(int ch1, int ch2)
226{
227 if (ch1 == 0) {
228 return (int16_t)ch2;
229 }
230
231 if (ch2 == 0) {
232 return (int16_t)ch1;
233 }
234
235 if ((ch1 > 0 && ch2 < 0) || (ch1 < 0 && ch2 > 0)) {
236 return (int16_t)(ch1 + ch2);
237 }
238
239 if (ch1 > 0) {
240 return (int16_t)((ch1 + ch2) - (ch1 * ch2 / 32768));
241 }
242
243 return (int16_t)-((-(ch1) + -(ch2)) - (-(ch1) * -(ch2) / 32768));
244}
245#endif
