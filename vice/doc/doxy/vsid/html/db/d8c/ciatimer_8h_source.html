155_CIAT_FUNC void ciat_set_alarm(ciat_t *state, CLOCK cclk)
156{
157 CLOCK tmp = 0;
158 CLOCK aclk = state->clk;
159 uint16_t cnt = state->cnt;
160 ciat_tstate_t t = state->state;
161
162 CIAT_LOGIN(("%s set_alarm: cclk=%lu, latch=%d",
163 state->name, cclk, state->latch));
164
165 while (1) {
166 CIAT_LOG(("- state->clk=%lu cnt=%d state=%04x", aclk, cnt, t));
167
168 if (((t & (CIAT_CR_START | CIAT_CR_FLOAD | CIAT_LOAD1
169 | CIAT_PHI2IN | CIAT_COUNT2 | CIAT_COUNT3 | CIAT_COUNT
170 | CIAT_LOAD))
171 == (CIAT_CR_START | CIAT_PHI2IN | CIAT_COUNT2 | CIAT_COUNT3
172 | CIAT_COUNT))
173 && (((t & CIAT_CR_ONESHOT) && (t & CIAT_ONESHOT0)
174 && (t & CIAT_ONESHOT))
175 || ((!(t & CIAT_CR_ONESHOT)) && (!(t & CIAT_ONESHOT0))
176 && (!(t & CIAT_ONESHOT))))) {
177 /* warp counting */
178 tmp = aclk + cnt;
179 break;
180 } else
181 if ((!(t & (CIAT_COUNT2 | CIAT_COUNT3 | CIAT_COUNT)))
182 && ((!(t & CIAT_CR_START))
183 || (!(t & (CIAT_PHI2IN | CIAT_STEP))))
184 && (((t & CIAT_CR_ONESHOT) && (t & CIAT_ONESHOT0)
185 && (t & CIAT_ONESHOT))
186 || ((!(t & CIAT_CR_ONESHOT)) && (!(t & CIAT_ONESHOT0))
187 && (!(t & CIAT_ONESHOT))))) {
188 /* warp stopped */
189 tmp = CLOCK_MAX;
190 break;
191 } else {
192 /* inc */
193 if (cnt && (t & CIAT_COUNT3)) {
194 cnt--;
195 }
196 t = ciat_table[t];
197 aclk++;
198 }
199
200 if ((cnt == 0) && (t & CIAT_COUNT3)) {
201 t |= CIAT_LOAD | CIAT_OUT;
202 tmp = aclk;
203 break;
204 }
205 if (t & CIAT_LOAD) {
206 cnt = state->latch;
207 t &= ~CIAT_COUNT3;
208 }
209 if ((t & CIAT_OUT)
210 && (t & (CIAT_ONESHOT | CIAT_ONESHOT0))) {
211 t &= ~(CIAT_CR_START | CIAT_COUNT2);
212 }
213 }
214/*
215 CIAT_LOG(("-> state->clk=%d cnt=%d, latch=%d, state=%04x",
216 state->clk, state->cnt, state->latch, t));
217*/
218 CIAT_LOG((" -> alarmclk=%d", tmp));
219
220 state->alarmclk = tmp;
221 if (tmp != CLOCK_MAX) {
222 alarm_set(state->alarm, tmp);
223 } else {
224 alarm_unset(state->alarm);
225 }
226
227 CIAT_LOGOUT((""));
228}

232 return state->alarmclk;

236_CIAT_FUNC int ciat_update(ciat_t *state, CLOCK cclk)
237{
238 int n;
239 CLOCK m;
240 ciat_tstate_t t = state->state;
241
242/* printf("%s update: state->clk=%d cclk=%d, state=%d, cnt=%d, latch=%d\n",
243 state->name, state->clk, cclk, t, state->cnt, state->latch); */
244
245#if 0
246 if (cclk > clk && cclk - clk > 0x10000) {
247 printf("ciat_update(%s: myclk=%d, cclk=%d)\n", state->name, clk, cclk);
248 }
249 if (cclk < state->clk) {
250 /* should never happen! */
251 printf("clock handling f*cked up, timer %s, tclk=%d, cpuclk=%d\n",
252 state->name, state->clk, cclk);
253 }
254#endif
255
256 n = 0;
257
258 CIAT_LOGIN(("%s update: cclk=%lu, latch=%d",
259 state->name, cclk, state->latch));
260
261 while (state->clk < cclk) {
262 CIAT_LOG(("- clk=%lu cnt=%d state=%04x", state->clk, state->cnt, t));
263
264 if (((t & (CIAT_CR_START | CIAT_CR_FLOAD | CIAT_LOAD1
265 | CIAT_PHI2IN | CIAT_COUNT2 | CIAT_COUNT3 | CIAT_COUNT
266 | CIAT_LOAD))
267 == (CIAT_CR_START | CIAT_PHI2IN | CIAT_COUNT2 | CIAT_COUNT3
268 | CIAT_COUNT))
269 && (((t & CIAT_CR_ONESHOT) && (t & CIAT_ONESHOT0)
270 && (t & CIAT_ONESHOT))
271 || ((!(t & CIAT_CR_ONESHOT)) && (!(t & CIAT_ONESHOT0))
272 && (!(t & CIAT_ONESHOT))))) {
273 /* warp counting */
274 if (state->clk + state->cnt > cclk) {
275 state->cnt -= ((uint16_t)(cclk - state->clk));
276 state->clk = cclk;
277 } else {
278 if (t & (CIAT_CR_ONESHOT | CIAT_ONESHOT0)) {
279 state->clk = state->clk + state->cnt;
280 state->cnt = 0;
281 } else {
282 /* overflow clk <= cclk */
283 state->clk = state->clk + state->cnt;
284 state->cnt = 0;
285 /* n++; */
286 if (((uint16_t)(cclk - state->clk)) >= state->latch + 1) {
287 m = (cclk - state->clk) / (state->latch + 1);
288 n += m;
289 state->clk += m * (state->latch + 1);
290 }
291 }
292 /* here we have cnt=0 and clk <= cclk */
293 }
294 } else
295 if ((!(t & (CIAT_COUNT2 | CIAT_COUNT3 | CIAT_COUNT)))
296 && ((!(t & CIAT_CR_START))
297 || (!(t & (CIAT_PHI2IN | CIAT_STEP))))
298 && (!(t & (CIAT_CR_FLOAD | CIAT_LOAD1 | CIAT_LOAD)))
299 && (((t & CIAT_CR_ONESHOT) && (t & CIAT_ONESHOT0)
300 && (t & CIAT_ONESHOT))
301 || ((!(t & CIAT_CR_ONESHOT)) && (!(t & CIAT_ONESHOT0))
302 && (!(t & CIAT_ONESHOT))))) {
303 /* warp stopped */
304 state->clk = cclk;
305 } else
306 if ((t == (CIAT_COUNT | CIAT_OUT | CIAT_LOAD | CIAT_PHI2IN
307 | CIAT_COUNT2 | CIAT_CR_START))
308 && (state->cnt == 1)
309 && (state->latch == 1)) {
310 /* when latch=1 and cnt=1 this warps up to clk */
311 m = (int)((cclk - state->clk) & (CLOCK) ~1);
312 if (m) {
313 state->clk += m;
314 n += (m >> 1);
315 } else {
316 t = ciat_table[t];
317 state->clk++;
318 }
319 } else {
320 /* inc */
321 if (state->cnt && (t & CIAT_COUNT3)) {
322 state->cnt--;
323 }
324 t = ciat_table[t];
325 state->clk++;
326 }
327
328 if ((state->cnt == 0) && (t & CIAT_COUNT3)) {
329 t |= CIAT_LOAD | CIAT_OUT;
330 n++;
331 }
332 if (t & CIAT_LOAD) {
333 state->cnt = state->latch;
334 t &= ~CIAT_COUNT3;
335 }
336 if ((t & CIAT_OUT)
337 && (t & (CIAT_ONESHOT | CIAT_ONESHOT0))) {
338 t &= ~(CIAT_CR_START | CIAT_COUNT2);
339 }
340 }
341
342 state->state = t;
343
344 CIAT_LOG(("-> state->clk=%lu cnt=%d, latch=%d, state=%04x",
345 state->clk, state->cnt, state->latch, t));
346
347 CIAT_LOGOUT(("-> n=%d", n));
348
349 return n; /* FIXME FIXME FIXME */
350}

392_CIAT_FUNC void ciat_set_latchhi(ciat_t *state, CLOCK cclk, uint8_t byte)
393{
394 CIAT_LOGIN(("%s set_latchhi: cclk=%lu, byte=%02x", state->name, cclk, byte));
395 state->latch = (state->latch & 0xff) | (byte << 8);
396 if ((state->state & CIAT_LOAD) || !(state->state & CIAT_CR_START)) {
397 state->cnt = state->latch;
398 }
399
400 /* just in case we have an underflow or force load and a scheduled alarm */
401 ciat_set_alarm(state, cclk);
402
403 CIAT_LOGOUT((""));
404}

406_CIAT_FUNC void ciat_set_latchlo(ciat_t *state, CLOCK cclk, uint8_t byte)
407{
408 CIAT_LOGIN(("%s set_latchlo(byte=%02x)", state->name, byte));
409
410 state->latch = (state->latch & 0xff00) | byte;
411 if (state->state & CIAT_LOAD) {
412 state->cnt = (state->cnt & 0xff00) | byte;

414
415 /* just in case we have an underflow or force load and a scheduled alarm */
416 ciat_set_alarm(state, cclk);

418 CIAT_LOGOUT((""));

423_CIAT_FUNC void ciat_set_ctrl(ciat_t *state, CLOCK cclk, uint8_t byte)
424{
425 CIAT_LOGIN(("%s set_ctrl: cclk=%lu, byte=%02x",
426 state->name, cclk, byte));
427
428 /* bit 0= start/stop, 3=oneshot 4=force load, 5=0:count phi2 1:singlestep */ state->state &= ~(CIAT_CR_MASK);
429 state->state |= (byte & CIAT_CR_MASK) ^ CIAT_PHI2IN;
430
431 ciat_set_alarm(state, cclk);
432
433 CIAT_LOGOUT((""));
434}
