85typedef struct cmdbus_s {

89 bit 7 is /PREADY
90 bit 6 is /PCLK

92 bit 4 is /PEXT

95 cmdbus_update is called)
96*/
97 uint8_t drv_data[NUM_DISK_UNITS];
98 uint8_t cpu_bus; /* same as drv_bus */
