43typedef struct cmdline_option_s {
44 /* Name of command-line option
45 *
46 * This cannot be `const`, since some code (see drive-cmdline-options.c
47 * for example) uses this field to store and later free dynamically
48 * allocated strings.
49 */
50 char *name;
51
52 /* Behavior of this command-line option. */
53 cmdline_option_type_t type;
54
55 /* Command line option flags:
56 CMDLINE_ATTRIB_NEED_ARGS: This option needs an argument.
57 CMDLINE_ATTRIB_NEED_BRACKETS: The brackets need to be added to the text.
58 CMDLINE_ATTRIB_DYNAMIC_DESCRIPTION: The description text is a pointer to a function that returns the text.
59 */
60 int attributes;
61
62 /* Function to call if type is `CALL_FUNCTION'. */
63 int (*set_func)(const char *value, void *extra_param);
64
65 /* Extra parameter to pass to `set_func' if type is `CALL_FUNCTION'. */
66 void *extra_param;
67
68 /* Resource to change if `type' is `SET_RESOURCE'
69 *
70 * Also cannot be `const`, see comment of `name` field
71 */
72 char *resource_name;
73
74 /* Value to assign to `resource_name' if `type' is `SET_RESOURCE' and
75 `need_arg' is zero. */
76 void *resource_value;
77
78 /* String to display after the option name in the help screen. */
79 const char *param_name;
80
81 /* Description string. */
82 const char *description;
83} cmdline_option_t;
