141inline static void interrupt_set_irq(interrupt_cpu_status_t *cs,
142 unsigned int int_num,
143 int value, CLOCK cpu_clk)
144{
145 if (cs == NULL || int_num >= cs->num_ints) {
146 return;
147 }
148
149 if (value) { /* Trigger the IRQ. */
150 if (!(cs->pending_int[int_num] & IK_IRQ)) {
151 cs->nirq++;
152 cs->global_pending_int = (cs->global_pending_int | (unsigned int)(IK_IRQ | IK_IRQPEND));
153 cs->pending_int[int_num] = (cs->pending_int[int_num] | (unsigned int)IK_IRQ);
154
155 cs->irq_pending_clk = CLOCK_MAX;
156
157 /* This makes sure that IRQ delay is correctly emulated when
158 cycles are stolen from the CPU. */
159#ifdef DEBUG
160 if (debug.maincpu_traceflg) {
161 log_debug("ICLK=%lu last_stolen_cycle=%lu",
162 (unsigned long)cpu_clk,
163 (unsigned long)(cs->last_stolen_cycles_clk));
164 }
165#endif
166 cs->irq_delay_cycles = 0;
167
168 if (cs->last_stolen_cycles_clk <= cpu_clk) {
169 cs->irq_clk = cpu_clk;
170 } else {
171 interrupt_fixup_int_clk(cs, cpu_clk, &(cs->irq_clk));
172 }
173 }
174 } else { /* Remove the IRQ condition. */
175 if (cs->pending_int[int_num] & IK_IRQ) {
176 if (cs->nirq > 0) {
177 cs->pending_int[int_num] =
178 (cs->pending_int[int_num] & (unsigned int)~IK_IRQ);
179 if (--cs->nirq == 0) {
180 cs->global_pending_int =
181 (cs->global_pending_int & (unsigned int)~IK_IRQ);
182 cs->irq_pending_clk = cpu_clk + 3;
183 }
184 } else {
185 interrupt_log_wrong_nirq();
186 }
187 }
188 }
189}

192inline static void interrupt_set_nmi(interrupt_cpu_status_t *cs,
193 unsigned int int_num,
194 int value, CLOCK cpu_clk)
195{
196 if (cs == NULL || int_num >= cs->num_ints) {
197 return;
198 }
199
200 if (value) { /* Trigger the NMI. */
201 if (!(cs->pending_int[int_num] & IK_NMI)) {
202 if (cs->nnmi == 0 && !(cs->global_pending_int & IK_NMI)) {
203 cs->global_pending_int = (cs->global_pending_int | IK_NMI);
204
205#ifdef DEBUG
206 if (debug.maincpu_traceflg) {
207 log_debug("ICLK=%lu last_stolen_cycle=%lu",
208 (unsigned long)cpu_clk,
209 (unsigned long)(cs->last_stolen_cycles_clk));
210 }
211#endif
212 /* This makes sure that NMI delay is correctly emulated when
213 cycles are stolen from the CPU. */
214 cs->nmi_delay_cycles = 0;
215
216 if (cs->last_stolen_cycles_clk <= cpu_clk) {
217 cs->nmi_clk = cpu_clk;
218 } else {
219 interrupt_fixup_int_clk(cs, cpu_clk, &(cs->nmi_clk));
220 }
221 }
222 cs->nnmi++;
223 cs->pending_int[int_num] = (cs->pending_int[int_num] | IK_NMI);
224 }
225 } else { /* Remove the NMI condition. */
226 if (cs->pending_int[int_num] & IK_NMI) {
227 if (cs->nnmi > 0) {
228 cs->nnmi--;
229 cs->pending_int[int_num] =
230 (cs->pending_int[int_num] & (unsigned int)~IK_NMI);
231#if 0
232 /* It should not be possible to remove the NMI condition,
233 only interrupt_ack_nmi() should clear it. */
234 if (cpu_clk == cs->nmi_clk) {
235 cs->global_pending_int = (enum cpu_int)(cs->global_pending_int & ~IK_NMI);
236 }
237#endif
238 } else {
239 interrupt_log_wrong_nnmi();
240 }
241 }
242 }
243}

245/* Change the interrupt line state: this can be used to change both NMI
246 and IRQ lines. It is slower than `interrupt_set_nmi()' and
247 `interrupt_set_irq()', but is left for backward compatibility (it works
248 like the old `setirq()'). */
249inline static void interrupt_set_int(interrupt_cpu_status_t *cs, int int_num,
250 enum cpu_int value, CLOCK cpu_clk)
251{
252 interrupt_set_nmi(cs, (unsigned int)int_num, (int)(value & IK_NMI), cpu_clk);
253 interrupt_set_irq(cs, (unsigned int)int_num, (int)(value & IK_IRQ), cpu_clk);
254}

258/* This function must be called by the CPU emulator when a pending NMI/IRQ
259 request is served. */
260inline static void interrupt_ack_nmi(interrupt_cpu_status_t *cs)
261{
262 cs->global_pending_int =
263 (cs->global_pending_int & (unsigned int)~IK_NMI);
264
265 if (cs->nmi_trap_func) {
266 cs->nmi_trap_func();
267 }
268}

270inline static void interrupt_ack_irq(interrupt_cpu_status_t *cs)
271{
272 cs->global_pending_int =
273 (cs->global_pending_int & (unsigned int)~IK_IRQPEND);
274 cs->irq_pending_clk = CLOCK_MAX;
275}
