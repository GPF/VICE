89struct draw_buffer_s {
90 /* The real drawing buffers, with padding bytes on either side to workaround CRT and Scale2x bugs */
91 uint8_t *draw_buffer_padded_allocations[2];
92 /* Pointers into the non padded buffers within the padded buffers */

94 /* The memory buffer where the screen of the emulated machine is drawn. Palettized, 1 byte per pixel */
95 uint8_t *draw_buffer;
96 /* Width of draw_buffer in pixels */
97 unsigned int draw_buffer_width;
98 /* Height of draw_buffer in pixels. Typically same as geometry->screen_size.height */
99 unsigned int draw_buffer_height;
100 unsigned int draw_buffer_pitch;
101 /* Width of emulator screen (physical screen on the machine where the emulator runs) in pixels */
102 unsigned int canvas_physical_width;
103 /* Height of emulator screen (physical screen on the machine where the emulator runs) in pixels */
104 unsigned int canvas_physical_height;
105 /* Maximum theoretical width of draw_buffer that would fit in the emulator screen.
106 Typically, it is the same as canvas_physical_width if no horizontal stretch is used (videoconfig->scalex == 1) and smaller if it is used.
107 TODO do we really need it? */
108 unsigned int canvas_width;
109 /* Maximum theoretical height of draw_buffer that would fit in the emulator screen.
110 Typically, it is the same as canvas_physical_width if no vertical stretch is used (videoconfig->scaley == 1) and smaller if it is used.
111 TODO do we really need it? */
112 unsigned int canvas_height;
113 /* Width of the visible subset of draw_buffer, in pixels. Typically same as geometry->screen_size.width */
114 unsigned int visible_width;

116 unsigned int visible_height;
117};
