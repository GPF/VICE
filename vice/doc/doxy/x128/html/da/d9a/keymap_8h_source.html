82enum shift_type {
83 NO_SHIFT = 0, /* Key is not shifted. Keys will be deshifted,
84 no other flags will be checked */
85
86 VIRTUAL_SHIFT = (1 << 0), /* The key needs a shift on the emulated machine. */
87 LEFT_SHIFT = (1 << 1), /* Key is left shift on the emulated machine. */
88 RIGHT_SHIFT = (1 << 2), /* Key is right shift on the emulated machine. */
89 ALLOW_SHIFT = (1 << 3), /* Allow key to be shifted. */
90 DESHIFT_SHIFT = (1 << 4), /* Although SHIFT might be pressed, do not
91 press shift on the emulated machine. */
92 ALLOW_OTHER = (1 << 5), /* Allow another key code to be assigned if
93 SHIFT is pressed. */
94 SHIFT_LOCK = (1 << 6), /* Key is shift lock on the emulated machine */
95 MAP_MOD_SHIFT = (1 << 7), /* Key requires SHIFT to be pressed on host */
96
97 ALT_MAP = (1 << 8), /* Key is used for an alternative keyboard mapping (x128) */
98
99 MAP_MOD_RIGHT_ALT = (1 << 9), /* Key requires right ALT (Alt-gr) to be pressed on host */
100 MAP_MOD_CTRL = (1 << 10), /* Key requires control to be pressed on host */
101
102 VIRTUAL_CBM = (1 << 11), /* The key is combined with CBM on the emulated machine */
103 VIRTUAL_CTRL = (1 << 12), /* The key is combined with CTRL on the emulated machine */
104
105 LEFT_CBM = (1 << 13), /* Key is CBM on the emulated machine */
106 LEFT_CTRL = (1 << 14), /* Key is CTRL on the emulated machine */
107
108 KEYFLG_NO_LOCK = (1 << 15), /* do NOT emulate a "locked" key for this key */
109
110 /* used in keyboard emulation code, not in keymaps */
111 IS_PRESSED = (1 << 30) /* is pressed */
112};

236static inline int key_is_modifier(int row, int column) {
237 if ((rshift_defined() && (row == kbd_rshiftrow) && (column == kbd_rshiftcol)) ||
238 (lshift_defined() && (row == kbd_lshiftrow) && (column == kbd_lshiftcol)) ||
239 (lcbm_defined() && (row == kbd_lcbmrow) && (column == kbd_lcbmcol)) ||
240 (lctrl_defined() && (row == kbd_lctrlrow) && (column == kbd_lctrlcol))) {
241 return 1;
242 }
243 return 0;
244}
