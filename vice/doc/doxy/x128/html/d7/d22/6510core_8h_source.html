28#ifndef VICE_6510CORE_H
29#define VICE_6510CORE_H
30
32#define OPINFO_DELAYS_INTERRUPT_MSK (1 << 8)
33#define OPINFO_DISABLES_IRQ_MSK (1 << 9)
34#define OPINFO_ENABLES_IRQ_MSK (1 << 10)
35
37#define OPINFO_DELAYS_INTERRUPT(opinfo) \
38 ((opinfo) & OPINFO_DELAYS_INTERRUPT_MSK)
39
40/* Return nonzero if `opinfo' has changed the I flag from 0 to 1, so that an
41 IRQ that happened 2 or more cycles before the end of the opcode should be
42 allowed. */
43#define OPINFO_DISABLES_IRQ(opinfo) \
44 ((opinfo) & OPINFO_DISABLES_IRQ_MSK)
45
46/* Return nonzero if `opinfo' has changed the I flag from 1 to 0, so that an
47 IRQ that happened 2 or more cycles before the end of the opcode should not
48 be allowed. */
49#define OPINFO_ENABLES_IRQ(opinfo) \
50 ((opinfo) & OPINFO_ENABLES_IRQ_MSK)
51
52/* Set the information for `opinfo'. `number' is the opcode number,
53 `delays_interrupt' must be non-zero if it causes a 1-cycle interrupt
54 delay, `disables_interrupts' must be non-zero if it disabled IRQs. */
55#define OPINFO_SET(opinfo, \
56 number, delays_interrupt, disables_irq, enables_irq) \
57 ((opinfo) = ((number) \
58 | ((delays_interrupt) ? OPINFO_DELAYS_INTERRUPT_MSK : 0) \
59 | ((disables_irq) ? OPINFO_DISABLES_IRQ_MSK : 0) \
60 | ((enables_irq) ? OPINFO_ENABLES_IRQ_MSK : 0)))
61
62/* Set whether the opcode causes the 1-cycle interrupt delay according to
63 `delay'. */
64#define OPINFO_SET_DELAYS_INTERRUPT(opinfo, delay) \
65 do { \
66 if ((delay)) { \
67 (opinfo) |= OPINFO_DELAYS_INTERRUPT_MSK; \
68 } \
69 } while (0)
70
71/* Set whether the opcode disables previously enabled IRQs according to
72 `disable'. */
73#define OPINFO_SET_DISABLES_IRQ(opinfo, disable) \
74 do { \
75 if ((disable)) { \
76 (opinfo) |= OPINFO_DISABLES_IRQ_MSK; \
77 } \
78 } while (0)
79
80/* Set whether the opcode enables previously disabled IRQs according to
81 `enable'. */
82#define OPINFO_SET_ENABLES_IRQ(opinfo, enable) \
83 do { \
84 if ((enable)) { \
85 (opinfo) |= OPINFO_ENABLES_IRQ_MSK; \
86 } \
87 } while (0)
88
89#endif
