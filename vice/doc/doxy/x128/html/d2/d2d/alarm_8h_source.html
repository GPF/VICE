110inline static void alarm_context_update_next_pending(alarm_context_t *context)
111{
112 CLOCK next_pending_alarm_clk = CLOCK_MAX;
113 int next_pending_alarm_idx;
114 unsigned int i;
115
116 next_pending_alarm_idx = context->next_pending_alarm_idx;
117
118 for (i = 0; i < context->num_pending_alarms; i++) {
119 CLOCK pending_clk = context->pending_alarms[i].clk;
120
121 if (pending_clk <= next_pending_alarm_clk) {
122 next_pending_alarm_clk = pending_clk;
123 next_pending_alarm_idx = (int)i;
124 }
125 }
126
127 context->next_pending_alarm_clk = next_pending_alarm_clk;
128 context->next_pending_alarm_idx = next_pending_alarm_idx;
129}

131inline static void alarm_context_dispatch(alarm_context_t *context,
132 CLOCK cpu_clk)
133{
134 CLOCK offset;
135 int idx;
136 alarm_t *alarm;
137
138 offset = cpu_clk - context->next_pending_alarm_clk;
139
140 idx = context->next_pending_alarm_idx;
141 alarm = context->pending_alarms[idx].alarm;
142
143 (alarm->callback)(offset, alarm->data);
144}

146inline static void alarm_set(alarm_t *alarm, CLOCK cpu_clk)
147{
148 alarm_context_t *context;
149 int idx;
150
151 context = alarm->context;
152 idx = alarm->pending_idx;
153
154 if (idx < 0) {
155 int new_idx;
156
157 /* Not pending yet: add. */
158
159 new_idx = (int)(context->num_pending_alarms);
160 if (new_idx >= (int)ALARM_CONTEXT_MAX_PENDING_ALARMS) {
161 alarm_log_too_many_alarms();
162 return;
163 }
164
165 context->pending_alarms[new_idx].alarm = alarm;
166 context->pending_alarms[new_idx].clk = cpu_clk;
167
168 context->num_pending_alarms++;
169
170 if (cpu_clk < context->next_pending_alarm_clk) {
171 context->next_pending_alarm_clk = cpu_clk;
172 context->next_pending_alarm_idx = new_idx;
173 }
174
175 alarm->pending_idx = new_idx;
176 } else {
177 /* Already pending: modify. */
178
179 context->pending_alarms[idx].clk = cpu_clk;
180 if (context->next_pending_alarm_clk > cpu_clk
181 || idx == context->next_pending_alarm_idx) {
182 alarm_context_update_next_pending(context);
183 }
184 }
185}
