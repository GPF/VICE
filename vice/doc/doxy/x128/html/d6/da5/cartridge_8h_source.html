267/* list of canonical names for the c64 cartridges:
268 note: often it is hard to determine "the" official name, let alone the way it
269 should be capitalized. because of that we go by the following rules:
270 - if even the actual spelling and/or naming is unclear, then the most "common"
271 variant is chosen ("Expert Cartridge" vs "The Expert")
272 - in many cases the name is printed all uppercase both on screen and in other
273 sources (manual, adverts). we refrain from doing the same here and convert
274 to Camel Case ("ACTION REPLAY V5" -> "Action Replay V5"), *except* if the
275 cart name constitutes an actual name (as in noun) by itself ("ISEPIC", "EXOS").
276 additionally common abbreviations such as RAM or EPROM will get written uppercase
277 if in doubt.
278 - although generally these cartridge names should never get translated, some
279 generic stuff is translated to english ("EPROM Karte" -> "EPROM Cart")

468/*
469 * VIC20 Generic cartridges
470 *
471 * The cartridge types below are only used during attach requests.
472 * They will always be converted to CARTRIDGE_VIC20_GENERIC when
473 * attached. This also means they can be remapped at will.
474 *
475 * bit 0: uses block 5 (a000-bfff)
476 * 1: uses block 3 (6000-7fff)
477 * 2: uses block 2 (4000-5fff)
478 * 3: uses block 1 (2000-3fff)
479 *
480 * bit 4+5: 0,0=1k, 0,1=2k 1,0=4k 1,1=8k (only useful if just one block is used)
481 * bit 6: starts at 0=first, 1=second 4k in block (only useful for 1k/2k/4k images)
482 */
483#define CARTRIDGE_VIC20_DETECT 0x8000
484
485#define CARTRIDGE_VIC20_TYPEDEF(h, s1, s0, b1, b2, b3, b5) \
486 (0x8000 | ((h) << 6) | ((s1) << 5) | ((s0) << 4) | ((b1) << 3) | ((b2) << 2) | ((b3) << 1) | ((b5) << 0))
487
