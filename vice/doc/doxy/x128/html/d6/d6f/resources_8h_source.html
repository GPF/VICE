40typedef enum resource_event_relevant_s {
41 /* The value can be different on client and server, and may change at any
42 time on either side. This is the default for most resources. */
43 RES_EVENT_NO,
44 /* Must be the same on server and client, when a value changes that event
45 is distributed to the other emulator. when client connects to server, the
46 server sends the initial values to the client.
47 example: MachineVideoStandard */
48 RES_EVENT_SAME,
49 /* must be the same on server and client, the value can not be changed.
50 Server sets the resource to event_strict_value when server is started.
51 Client sets the resource to event_strict_value when it connects to the
52 server.
53 example: InitialWarpMode = 0 */
54 RES_EVENT_STRICT
55} resource_event_relevant_t;

67struct resource_int_s {
68 /* Resource name
69 *
70 * This cannot be `const`, since some code uses this field to store, and
71 * later free, dynamically allocated strings. See for example
72 * video-resources.c
73 */

75
76 /* Factory default value. This cannot be 'const' since in some cases the
77 value will be determined and set at runtime */
78 int factory_value;
79
80 /* Flag: Is the resource important for history recording or netplay? */
