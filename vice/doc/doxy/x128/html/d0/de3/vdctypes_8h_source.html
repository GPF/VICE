191 /* used to record the value of the cpu clock at the start of a raster line */
192 CLOCK vdc_line_start;
193 /* based on blacky_stardust calculations, calculating current_x_pixel should be like:
194 current_x_pixel = pixels_per_line / (vdc.xsync_increment >> 16) * (current_cycle - vdc_line_start) */
195
196 /* record register 27 in case of a change between raster updates */
197 /* FIXME - never used! */
198 int old_reg27;
199
200 /* Row counter (required for comparison with reg[6] - number of visible screen rows - to know if we are at the end of the visible data) */
201 unsigned int row_counter;
202
203 /* Row counter_y counts individual raster lines of the current row to know if we are at the end of the current row. */
204 unsigned int row_counter_y;
205
206 /* offset into the attribute memory - used for emulating the 8x1 attribute VDC quirk */
207 unsigned int attribute_offset;
208
209 /* flag as to whether the VDC is actually rendering active raster lines, or is idle (i.e in the top or bottom border) */
210 unsigned int display_enable;
211
212 /* flag as to whether the VDC is preparing to draw but hasn't actually started yet, usually when it's reset to internal row # 0 */
213 unsigned int prime_draw;
214
215 /* flag as to whether the VDC is drawing (e.g. reading from screen memory) or if it is stopped. This is different to above, because the VDC can be drawing inside the top or bottom border */
216 unsigned int draw_active;
217
218 /* flag as to whether the VDC is finished drawing or not, to make sure end of frame stuff like video pointers is handled properly */
219 unsigned int draw_finished;
220
221 /* flag as to whether the VDC is in VSYNC or not */
222 unsigned int vsync;
223
224 /* internal VDC counter so the vsync is the correct height (in rows) */
225 unsigned int vsync_counter;
226
227 /* height of the vsync pulse in raster lines, this varies by video standard/video mode */
228 unsigned int vsync_height;
229
230 /* Row counter used by drawing code. Separate from row_counter above because the drawing can be asynch with the main video signal */
231 unsigned int draw_counter;
232
233 /* Raster line counter used by drawing code as above. */
234 unsigned int draw_counter_y;
235
236 /* used to monitor for changes in screen and/or attribute addresses, as the cache can't cope with that */
237 unsigned int old_screen_adr, old_attribute_adr;
238
239 /* Interlace flag - 0 if normal/non-interlaced, 1 if interlaced */
240 unsigned int interlaced;
241
242 /* Light pen. */
243 vdc_light_pen_t light_pen;
244
245 /* Size of vdc canvas, i.e. total visible screen area in pixels */
246 unsigned int canvas_width;
247 unsigned int canvas_height;
248
249 /* To compare if it changes since last frame */
250 unsigned int canvas_width_old;
251 unsigned int canvas_height_old;
252
253 /* Internal character and attribute buffers */
254 uint8_t scrnbuf[0x200];
255 unsigned int scrnbufdraw;
256 uint8_t attrbuf[0x200];
257 unsigned int attrbufdraw;
258};
