34/* This contains some redundancy to make it compatible with the way registers
35 are handled within the CPU emulation. The struct should be accessed using
36 the `MOS6510_REGS_*()' macros. */
37typedef struct mos6510_regs_s {
40 unsigned int pc;
41 uint8_t a;
42 uint8_t x;
43 uint8_t y;
44 uint8_t sp;
45 uint8_t p;
46 uint8_t n;
47 uint8_t z;
48} mos6510_regs_t;

50/* These define the position of the status flags in the P (`status')
51 register. */
52#define P_SIGN 0x80
53#define P_OVERFLOW 0x40
54#define P_UNUSED 0x20
55#define P_BREAK 0x10
56#define P_DECIMAL 0x08
57#define P_INTERRUPT 0x04
58#define P_ZERO 0x02
59#define P_CARRY 0x01

67#define MOS6510_REGS_SET_OVERFLOW(regs, val) \
68 do { \
69 if (val) { \

73 } \
74 } while (0)

112#define MOS6510_REGS_SET_SIGN(regs, val) ((regs)->n = (val) ? 0x80 : 0)

114#define MOS6510_REGS_SET_ZERO(regs, val) ((regs)->z = !(val))

116#define MOS6510_REGS_SET_STATUS(regs, val) \
117 ((regs)->p = ((val) & ~(P_ZERO | P_SIGN)), \
118 MOS6510_REGS_SET_ZERO(regs, (val) & P_ZERO), \
119 (regs)->n = (val))
